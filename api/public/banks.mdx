## Overview

Get a list of all publicly available banks supported by the Valour platform. This endpoint provides basic bank information that can be accessed without full authentication, making it suitable for public-facing applications and initial setup flows.

<Note>
This endpoint requires API key authentication. Include your API key in the `X-API-Key` header.
</Note>

## Usage Examples

### Get Public Banks for Dropdown

<CodeGroup>

```javascript JavaScript
async function getPublicBanksForDropdown(apiKey) {
  try {
    const response = await fetch('https://api.railsfromthecrypt.com/v1/public/banks', {
      method: 'GET',
      headers: {
        'X-API-Key': apiKey
      }
    });

    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    const data = await response.json();
    
    // Format for dropdown usage
    const bankOptions = data.data.results
      .filter(bank => bank.is_active)
      .map(bank => ({
        value: bank.bank_code,
        label: bank.bank_name,
        country: bank.country,
        currency: bank.currency
      }));
    
    return bankOptions;
  } catch (error) {
    console.error('Failed to fetch public banks:', error);
    throw error;
  }
}

// Usage
const bankOptions = await getPublicBanksForDropdown('your-api-key-here');
console.log('Bank options for dropdown:', bankOptions);
```

```python Python
def get_public_banks_for_dropdown(api_key):
    try:
        response = requests.get(
            'https://api.railsfromthecrypt.com/v1/public/banks',
            headers={'X-API-Key': api_key}
        )
        
        response.raise_for_status()
        data = response.json()
        
        # Format for dropdown usage
        bank_options = [
            {
                'value': bank['bank_code'],
                'label': bank['bank_name'],
                'country': bank['country'],
                'currency': bank['currency']
            }
            for bank in data['data']['results']
            if bank['is_active']
        ]
        
        return bank_options
        
    except requests.exceptions.RequestException as e:
        print(f'Failed to fetch public banks: {e}')
        raise

# Usage
bank_options = get_public_banks_for_dropdown('your-api-key-here')
print('Bank options for dropdown:', bank_options)
```

### Cache Public Banks

<CodeGroup>

```javascript JavaScript
class BankCache {
  constructor(apiKey) {
    this.apiKey = apiKey;
    this.cache = null;
    this.cacheExpiry = null;
    this.cacheDuration = 24 * 60 * 60 * 1000; // 24 hours
  }

  async getBanks() {
    // Check if cache is still valid
    if (this.cache && this.cacheExpiry && Date.now() < this.cacheExpiry) {
      return this.cache;
    }

    // Fetch fresh data
    try {
      const response = await fetch('https://api.railsfromthecrypt.com/v1/public/banks', {
        method: 'GET',
        headers: {
          'X-API-Key': this.apiKey
        }
      });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const data = await response.json();
      
      // Update cache
      this.cache = data.data.results;
      this.cacheExpiry = Date.now() + this.cacheDuration;
      
      return this.cache;
    } catch (error) {
      console.error('Failed to fetch banks:', error);
      
      // Return cached data if available, even if expired
      if (this.cache) {
        console.warn('Using expired cache data');
        return this.cache;
      }
      
      throw error;
    }
  }

  getBankByCode(bankCode) {
    if (!this.cache) {
      throw new Error('No cached data available');
    }
    
    return this.cache.find(bank => bank.bank_code === bankCode);
  }
}

// Usage
const bankCache = new BankCache('your-api-key-here');
const banks = await bankCache.getBanks();
const specificBank = bankCache.getBankByCode('001');
```
